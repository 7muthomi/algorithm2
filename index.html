function insertionSort(arr) { for (let i = 1; i < arr.length; i++) { let key = arr[i]; let j = i - 1; // Compare key with elements before it and shift them one position ahead while (j >= 0 && arr[j] > key) { arr[j + 1] = arr[j]; j--; } // Insert key at its correct position arr[j + 1] = key; } return arr; } // Example usage: let numbers = [3, 0, 2, 5, -1, 4, 1]; let sortedNumbers = insertionSort(numbers); console.log(sortedNumbers); // Output: [-1, 0, 1, 2, 3, 4, 5]
